54       public function __construct
55       (
56           string $entity,
57           string $alias = null
58       )
59       {
60           $this->entity = $entity;
61           try {
62               $this->reflectionClass = $this->setEntityClass($entity);
63           } catch (ReflectionException $e) {
64               var_dump($e->getMessage());die;
65           }
66           $this->alias = $alias;
67   
68           $type = 'mysql';
69   
70           $this->config = new Config('config/env.yaml');
71           $databaseConfig = $this->config->getConfig('database');
72           $this->dsn['host'] = $databaseConfig['DB_HOST'];
73           $this->dsn['dbname'] = $databaseConfig['DB_NAME'];
74           $this->username = $databaseConfig['DB_USER'];
75           $this->password = $databaseConfig['DB_PASS'];
76   
77           try {
78               $this->dsnString = "{$type}:";
79               $i = 0;
80               foreach ($this->dsn as $key => $value) {
81                   $this->dsnString .= "{$key}={$value}";
82                   if (++$i !== count($this->dsn)) $this->dsnString .= ";";
83               }
84   
85               // Neue PDO Instanz statt extend PDO, damit nur die QueryBuilder Methoden direkt verfügbar sind.
86               $this->pdo = new PDO($this->dsnString, $this->username, $this->password);
87               $this->pdo->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);
88               $this->pdo->setAttribute(PDO::ATTR_DEFAULT_FETCH_MODE, PDO::FETCH_BOTH);
89               $this->pdo->setAttribute(PDO::ATTR_CASE, PDO::CASE_NATURAL);
90           } catch (PDOException $e) {
91               echo 'Exception abgefangen: ' . $e->getMessage() . "\n";
92           }
93       }

107      private function getColumns(): string
108      {
109          $entityProperties = $this->reflectionClass->getProperties();
110          $columns = false;
111          foreach ($entityProperties as $property) {
112              if(!$property->isPrivate())
113              {
114                  $propertyNameAsArray = preg_split('/(?=[A-Z])/', $property->getName());
115                  $propertyNameAsSnakeTail = strtolower(implode('_', $propertyNameAsArray));
116                  if(null !== $this->alias)
117                  {
118                      $columns .= "{$this->alias}.{$propertyNameAsSnakeTail} AS {$property->getName()},";
119                  } else {
120                      $columns .= "{$propertyNameAsSnakeTail} AS {$property->getName()},";
121                  }
122  
123              }
124          }
125          return $columns = rtrim($columns, ',');
126      }
127  
128      private function generateSnakeTailString(string $value): string
129      {
130          $valueAsArray = preg_split('/(?=[A-Z])/', $value);
131          return strtolower(ltrim($propertyNameAsSnakeTail = implode('_', $valueAsArray),'_'));
132      }

315      /**
316       * @return $this
317       */
318      public function getQuery(): QueryBuilder
319      {
320  
321          if(0 !== count($this->projection))
322          {
323              $this->query['projection'] = "SELECT " . implode(',',$this->projection);
324          } else {
325              $this->query['projection'] = "SELECT *";
326          }
327  
328          $this->query['table'] = "FROM {$this->getTableNameFromEntity()}";
329  
330          if(null !== $this->alias)
331          {
332              $this->query['table'] .= " $this->alias ";
333          }
334  
335          if (0 !== count($this->joins))
336          {
337              $this->query['joins'] = implode(' ',$this->joins);
338          }
339  
340          if (0 !== count($this->conditions))
341          {
342              $this->query['condition'] = "WHERE " . implode(' ',$this->conditions);
343          }
344  
345          if (0 !== count($this->groupBy))
346          {
347              $this->query['group'] = "GROUP BY " . implode(',',$this->groupBy);
348          }
349  
350          if (0 !== count($this->groupConditions))
351          {
352              $this->query['group_condition'] = "HAVING " . implode(' ',$this->groupConditions);
353          }
354  
355          if (0 !== count($this->orderBy))
356          {
357              $order = "ORDER BY ";
358              foreach ($this->orderBy as $key => $value)
359              {
360                  $value = strtoupper($value);
361                  $order .= "{$key} {$value},";
362              }
363              $this->query['order'] = rtrim($order,',');
364  
365          }
366  
367          $this->query['limit'] = $this->limit;
368  
369          $this->query['offset'] = $this->offset;
370  
371          $this->statement = $this->pdo->prepare( implode(' ',$this->query));
372  
373          foreach ($this->parameters as $key => $value)
374          {
375              $this->statement->bindValue(":{$key}", $value);
376          }
377  
378          return $this;
379      }
